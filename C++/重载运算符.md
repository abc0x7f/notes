# 结构体内重载运算符
```cpp
struct node {
	int x, y;
	
	node operator + (const node& A) {
		x += A.x;
		y += A.y;
		return *this; // ?
	}

	node operator = (const node& A) {
		x = A.x;
		y = A.y;
		return *this;
	}

	bool operator < (const node& A) {
		return x == A.x ? y < A.y : x < A.x;
	}

	bool operator == (const node& A) {
		return x == A.x && y == A.y;
	}
	friend istream& operator >> (istream& in, node& A) {
		in >> A.x >> A.y;
		return in;
	}

	friend ostream& operator << (ostream&, const node& A) {
		cout << A.x << " " << A.y << endl;
		return cout;
	}
} a[100]; 
```
